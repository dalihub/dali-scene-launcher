CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
PROJECT(dali-scene-launcher C CXX)

SET(dali-scene-launcher_VERSION_MAJOR 1)
SET(dali-scene-launcher_VERSION_MINOR 0)

SET(ROOT_SRC_DIR ${CMAKE_SOURCE_DIR}/../..)

SET(PREFIX ${CMAKE_INSTALL_PREFIX})

IF(DEFINED DALI_APP_DIR)
        SET(BINDIR ${DALI_APP_DIR}/bin)
ELSE()
        SET(BINDIR ${PREFIX}/bin)
ENDIF()

IF(DEFINED DALI_APP_RES_DIR)
        SET(APP_DATA_RES_DIR ${DALI_APP_RES_DIR})
ELSE()
        SET(APP_DATA_RES_DIR ${PREFIX}/share/com.samsung.dali-scene-launcher/resources)
ENDIF()

MESSAGE( STATUS "CMAKE_BUILD_TYPE: " ${CMAKE_BUILD_TYPE} )
if(CMAKE_BUILD_TYPE MATCHES Debug)
    MESSAGE("Setting variable: DEBUG_ENABLED")
    SET(DEBUG_ENABLED "1")
ENDIF()

SET(RESOURCE_DIR ${ROOT_SRC_DIR}/resources)
SET(LOCAL_IMAGES_DIR ${RESOURCE_DIR}/images)
SET(LOCAL_MODELS_DIR ${RESOURCE_DIR}/models)
SET(LOCAL_SHADERS_DIR ${RESOURCE_DIR}/shaders)

SET(IMAGES_DIR ${APP_DATA_RES_DIR}/images/)
SET(MODELS_DIR ${APP_DATA_RES_DIR}/models/)
SET(SHADERS_DIR ${APP_DATA_RES_DIR}/shaders/)

IF(NOT DEFINED LOCALE_DIR)
        SET(LOCALE_DIR ${PREFIX}/share/locale)
ENDIF()

SET(SCENE_LAUNCHER_IMAGE_DIR \\"${IMAGES_DIR}\\")
SET(SCENE_LAUNCHER_MODEL_DIR \\"${MODELS_DIR}\\")
SET(SCENE_LAUNCHER_SHADER_DIR \\"${SHADERS_DIR}\\")

INSTALL(DIRECTORY ${LOCAL_IMAGES_DIR}/ DESTINATION ${IMAGES_DIR})
INSTALL(DIRECTORY ${LOCAL_MODELS_DIR}/ DESTINATION ${MODELS_DIR})
INSTALL(DIRECTORY ${LOCAL_SHADERS_DIR}/ DESTINATION ${SHADERS_DIR})

SET(PKG_LIST dali-core
             dali-adaptor
             dali-toolkit )

INCLUDE(FindPkgConfig)
pkg_check_modules(REQUIRED_PKGS REQUIRED ${PKG_LIST})

FOREACH(flag ${REQUIRED_PKGS_CFLAGS})
        SET(REQUIRED_CFLAGS "${REQUIRED_CFLAGS} ${flag}")
ENDFOREACH(flag)

SET(DALI_SCENE_LAUNCHER_CFLAGS "-DSCENE_LAUNCHER_IMAGE_DIR=${SCENE_LAUNCHER_IMAGE_DIR} -DSCENE_LAUNCHER_MODEL_DIR=${SCENE_LAUNCHER_MODEL_DIR} -DSCENE_LAUNCHER_SHADER_DIR=${SCENE_LAUNCHER_SHADER_DIR} -fvisibility=hidden -DHIDE_DALI_INTERNALS")

#Create resources location file
CONFIGURE_FILE( resources-location.in ${ROOT_SRC_DIR}/scene-launcher/resources-location.cpp )

###########################################################################

SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${REQUIRED_CFLAGS} ${DALI_SCENE_LAUNCHER_CFLAGS} -Werror -Wall -fPIE -std=c++11")
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${CMAKE_C_FLAGS}")

INCLUDE_DIRECTORIES(${ROOT_SRC_DIR})

ADD_SUBDIRECTORY(scene-launcher)
